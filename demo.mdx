import { useState } from 'react'

# 🧪 Live Code Demo: Counter Component

This is a live, interactive React component embedded right in your docs using **MDX + Mintlify**.

---

## ⚛️ Interactive Counter

```jsx live
function Counter() {
  const [count, setCount] = useState(0)

  return (
    <div style={{ textAlign: 'center', padding: '1rem', border: '1px solid #ccc', borderRadius: '10px' }}>
      <h2>Count: {count}</h2>
      <button onClick={() => setCount(count + 1)}>➕ Increment</button>
      <button onClick={() => setCount(count - 1)} style={{ marginLeft: '1rem' }}>➖ Decrement</button>
    </div>
  )
}

export default Counter
